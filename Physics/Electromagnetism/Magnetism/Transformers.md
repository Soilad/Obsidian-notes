$$
\begin{align}
\vec B_1 = \vec B_2\\
\Phi_1 = \Phi_2\\
\frac{d\Phi_1}{dt} = \frac{d\Phi_2}{dt}\\
|V_1| = N_1\frac{d\Phi_1}{dt}\\
|V_2| = N_2\frac{d\Phi_2}{dt}\\
\implies \frac{V_1}{N_1} = \frac{V_2}{N_2}
\end{align}
